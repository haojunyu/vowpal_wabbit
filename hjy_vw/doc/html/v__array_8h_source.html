31 template<class T> inline void push(v_array<T>& v, const T &new_ele)
32 {
33  if(v.end == v.end_array)
34  {
35  size_t old_length = v.end_array - v.begin;
36  size_t new_length = 2 * old_length + 3;
37  // size_t new_length = old_length + 1;
38  v.begin = (T *)realloc(v.begin,sizeof(T) * new_length);
39  v.end = v.begin + old_length;
40  v.end_array = v.begin + new_length;
41  }
42  *(v.end++) = new_ele;
43 }

46 template<class T> void push_many(v_array<T>& v, const T* begin, size_t num)
47 {
48  if(v.end+num >= v.end_array)
49  {
50  size_t length = v.end - v.begin;
51  size_t new_length = max(2 * (size_t)(v.end_array - v.begin) + 3,
52  v.end - v.begin + num);
53  v.begin = (T *)realloc(v.begin,sizeof(T) * new_length);
54  v.end = v.begin + length;
55  v.end_array = v.begin + new_length;
56  }
57  memcpy(v.end, begin, num * sizeof(T));
58  v.end += num;
59 }
