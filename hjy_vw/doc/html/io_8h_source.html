33  if (space.end_array - endloaded == 0)
34  {
35  size_t offset = endloaded - space.begin;
36  reserve(space, 2 * (space.end_array - space.begin));
37  endloaded = space.begin+offset;
38  }
39  ssize_t num_read = read(f, endloaded, space.end_array - endloaded);
40  if (num_read >= 0)
41  {
42  endloaded = endloaded+num_read;
43  return num_read;
44  }
45  else
46  return 0;
47  }

50  cerr << "error, failed to write example\n";
